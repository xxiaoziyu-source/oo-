function gradient(text, startColor, endColor)
    local result = ""
    local length = #text
    
    for i = 1, length do
        local t = (i - 1) / math.max(length - 1, 1)
        local r = math.floor((startColor.R + (endColor.R - startColor.R) * t) * 255)
        local g = math.floor((startColor.G + (endColor.G - startColor.G) * t) * 255)
        local b = math.floor((startColor.B + (endColor.B - startColor.B) * t) * 255)

        local char = text:sub(i, i)
        result = result .. "<font color=\"rgb(" .. r ..", " .. g .. ", " .. b .. ")\">" .. char .. "</font>"
    end
    
    return result
end

    while true do
        local currentTimestamp = tick()
        local currentTimeTable = os.date("*t", currentTimestamp)
        local milliseconds = math.floor((currentTimestamp - math.floor(currentTimestamp)) * 1000)
        
        local hours = string.format("%02d", currentTimeTable.hour)
        local minutes = string.format("%02d", currentTimeTable.min)
        local seconds = string.format("%02d", currentTimeTable.sec)
        local ms = string.format("%03d", milliseconds)
        local weekDay = weekDays[currentTimeTable.wday]
        
        local realWorldTimeString = hours .. ":" .. minutes .. ":" .. seconds .. ":" .. ms .. " тЉе" .. weekDay
        
        TimeTag:SetTitle(realWorldTimeString)
        
        task.wait(0.01)
    end
end

local function translateText(text)
    if not text or type(text) ~= "string" then return text end
    if translationCache[text] then return translationCache[text] end
    if Translations[text] then
        translationCache[text] = Translations[text]
        return Translations[text]
    end
    for en, cn in pairs(Translations) do
        if text:find(en) then
            local result = text:gsub(en, cn)
            translationCache[text] = result
            return result
        end
    end
    translationCache[text] = text
    return text
end

local function logTranslation(original, translated)
    if original == translated then return end
    local logMsg = string.format("[%s] Рєњ [%s]", original, translated)
    if not table.find(notificationQueue, logMsg) then
        table.insert(notificationQueue, logMsg)
        while #notificationQueue > 10 do table.remove(notificationQueue, 1) end
        translatedCount = translatedCount + 1
    end
end

    end
end

local function startTranslationEngine()
    local success, err = pcall(function()
        local oldIndex = getrawmetatable(game).__newindex
        setreadonly(getrawmetatable(game), false)
        getrawmetatable(game).__newindex = newcclosure(function(t, k, v)
            if isTranslating and (t:IsA("TextLabel") or t:IsA("TextButton") or t:IsA("TextBox")) and k == "Text" then
                local original = tostring(v)
                scannedCount = scannedCount + 1
                local translated = translateText(original)
                logTranslation(original, translated)
                v = translated
            end
            return oldIndex(t, k, v)
        end)
        setreadonly(getrawmetatable(game), true)
    end)

    if not success then
        warn("тЁЃУАетіФТїЂтц▒У┤Ц№╝їтљ»ућетЁют║ЋТЅФТЈЈ:", err)
    end

    local translated = {}
    local function scanAndTranslate(root)
        if not root or not isTranslating then return end
        for _, gui in ipairs(root:GetDescendants()) do
            if (gui:IsA("TextLabel") or gui:IsA("TextButton") or gui:IsA("TextBox")) and not translated[gui] then
                pcall(function()
                    local original = gui.Text
                    if original and original ~= "" then
                        scannedCount = scannedCount + 1
                        local translatedText = translateText(original)
                        if translatedText ~= original then
                            gui.Text = translatedText
                            logTranslation(original, translatedText)
                            translated[gui] = true
                        end
                    end
                end)
            end
        end
    end

    local function setupListener(parent)
        local conn = parent.DescendantAdded:Connect(function(descendant)
            if isTranslating and (descendant:IsA("TextLabel") or descendant:IsA("TextButton") or descendant:IsA("TextBox")) then
                task.wait(0.1)
                pcall(function()
                    local original = descendant.Text
                    if original and original ~= "" then
                        scannedCount = scannedCount + 1
                        local translatedText = translateText(original)
                        if translatedText ~= original then
                            descendant.Text = translatedText
                            logTranslation(original, translatedText)
                        end
                    end
                end)
            end
        end)
        table.insert(connections, conn)
    end

    task.spawn(function()
        setupListener(CoreGui)
        setupListener(PlayerGui)
        while isTranslating do
            scanAndTranslate(CoreGui)
            scanAndTranslate(PlayerGui)
            updateStatus()
            updateNotifications()
            task.wait(2)
        end
    end)
end

local function stopTranslationEngine()
    isTranslating = false
    for _, conn in ipairs(connections) do conn:Disconnect() end
    connections = {}
    if notificationGui then notificationGui.Visible = false end
    translatedCount = 0
    scannedCount = 0
    notificationQueue = {}
    translationCache = {}
    updateStatus()
end
